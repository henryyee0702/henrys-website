class SolarPhysicsEngine {
  constructor(nodes, timingConfig, isReducedMotion) {
    this.nodes = nodes;
    this.timingConfig = timingConfig;
    this.subscribers = new Map();
    this.cameraSubscriber = null;
    this.isReducedMotion = isReducedMotion;
    
    this.state = { isPaused: false, hoveredId: null, targetId: null, activeId: null, phase: 'idle' };
    this.physics = { speedMultiplier: timingConfig.NORMAL_MULTIPLIER, camX: 0, camY: 0, nodes: {} };

    // 泛化：用半徑 (radius === 0) 判斷是否為中心星體，而非依賴 isSun 旗標
    const orbitNodeCount = nodes.filter(n => n.radius > 0).length;
    nodes.forEach((n, index) => {
      this.physics.nodes[n.id] = {
        z: 0, scale: 1, labelY: 0, labelScale: 1,
        angle: n.radius === 0 ? 0 : (index * 2 * Math.PI) / (orbitNodeCount || 1)
      };
    });

    this.lastTime = performance.now();
    this.reqId = null;
    this.loop = this.loop.bind(this);
  }

  updateContext(newState, isReducedMotion) {
    this.state = { ...this.state, ...newState };
    this.isReducedMotion = isReducedMotion;
  }

  subscribeNode(id, callback) {
    this.subscribers.set(id, callback);
    return () => this.subscribers.delete(id);
  }

  subscribeCamera(callback) {
    this.cameraSubscriber = callback;
    return () => { this.cameraSubscriber = null; };
  }

  start() {
    if (!this.reqId) this.reqId = requestAnimationFrame(this.loop);
  }

  stop() {
    if (this.reqId) {
      cancelAnimationFrame(this.reqId);
      this.reqId = null;
    }
  }

  loop(timestamp) {
    let deltaTime = timestamp - this.lastTime;
    this.lastTime = timestamp;
    if (deltaTime > 100) deltaTime = 16.666; 
    const timeScale = deltaTime / 16.666;

    const frameAwareLerp = (start, end, factor) => start + (end - start) * (1 - Math.pow(1 - factor, timeScale));

    const targetSpeedMultiplier = this.isReducedMotion ? 0 : 
      (this.state.phase === 'pullback' ? this.timingConfig.SPEED_UP_MULTIPLIER : this.timingConfig.NORMAL_MULTIPLIER);

    this.physics.speedMultiplier = frameAwareLerp(this.physics.speedMultiplier, targetSpeedMultiplier, this.timingConfig.LERP_FACTOR_SPEED);

    const isActuallyPlaying = !this.state.isPaused && !this.state.hoveredId && !this.isReducedMotion;
    if (isActuallyPlaying || this.physics.speedMultiplier > 1.05) { 
      this.nodes.forEach(node => {
        if (node.radius > 0) this.physics.nodes[node.id].angle += node.speed * this.physics.speedMultiplier * timeScale;
      });
    }

    let targetCamX = 0, targetCamY = 0;
    const isPushin = this.state.phase === 'pushin';
    const isZooming = this.state.phase !== 'idle';

    if (isPushin && this.state.targetId) {
      const targetNode = this.nodes.find(n => n.id === this.state.targetId);
      if (targetNode && targetNode.radius > 0) {
        const angle = this.physics.nodes[targetNode.id].angle;
        targetCamX = -targetNode.radius * Math.cos(angle);
        targetCamY = -targetNode.radius * Math.sin(angle);
      }
    }

    this.physics.camX = frameAwareLerp(this.physics.camX, targetCamX, this.timingConfig.LERP_FACTOR_CAM);
    this.physics.camY = frameAwareLerp(this.physics.camY, targetCamY, this.timingConfig.LERP_FACTOR_CAM);

    if (this.cameraSubscriber) this.cameraSubscriber(this.physics.camX, this.physics.camY);

    this.nodes.forEach(node => {
      const isTarget = this.state.targetId === node.id;
      const isSelected = this.state.activeId === node.id;
      const isHovered = this.state.hoveredId === node.id;
      const pNode = this.physics.nodes[node.id];

      let x = 0, y = 0;
      if (node.radius > 0) {
        x = node.radius * Math.cos(pNode.angle);
        y = node.radius * Math.sin(pNode.angle);
      }

      pNode.z = frameAwareLerp(pNode.z, (isTarget && isPushin) ? 100 : 0, this.timingConfig.LERP_FACTOR_Z);
      pNode.scale = frameAwareLerp(pNode.scale, ((isHovered || isSelected) && !isZooming) || (isTarget && isZooming) ? 1.15 : 1, 0.15);
      pNode.labelY = frameAwareLerp(pNode.labelY, ((isHovered || isSelected) && !isZooming) ? -8 : 0, 0.15);
      pNode.labelScale = frameAwareLerp(pNode.labelScale, ((isHovered || isSelected) && !isZooming) ? 1.1 : 1, 0.15);

      const notify = this.subscribers.get(node.id);
      if (notify) notify({ x, y, z: pNode.z, scale: pNode.scale, labelY: pNode.labelY, labelScale: pNode.labelScale });
    });

    this.reqId = requestAnimationFrame(this.loop);
  }
}

const EngineContext = createContext(null);

function useWindowContext() {
  const [windowWidth, setWindowWidth] = useState(typeof window !== 'undefined' ? window.innerWidth : 1200);
  const [scale, setScale] = useState(1);

  useEffect(() => {
    const handleResize = () => {
      const w = window.innerWidth;
      setWindowWidth(w);
      setScale(Math.min(1.1, Math.min(w, window.innerHeight) / CONFIG.LAYOUT.SYS_MAX_SIZE));
    };
    window.addEventListener('resize', handleResize);
    handleResize();
    return () => window.removeEventListener('resize', handleResize);
  }, []);

  return { windowWidth, scale };
}

function useReducedMotion() {
  const [reducedMotion, setReducedMotion] = useState(false);
  useEffect(() => {
    if (typeof window === 'undefined') return;
    const mediaQuery = window.matchMedia('(prefers-reduced-motion: reduce)');
    setReducedMotion(mediaQuery.matches);
    const handler = (e) => setReducedMotion(e.matches);
    mediaQuery.addEventListener('change', handler);
    return () => mediaQuery.removeEventListener('change', handler);
  }, []);
  return reducedMotion;
}

function useDialogFocus(isOpen, onClose) {
  const dialogRef = useRef(null);
  const previousFocusRef = useRef(null);

  useEffect(() => {
    if (isOpen) {
      previousFocusRef.current = document.activeElement;
      const focusTimer = setTimeout(() => {
        if (dialogRef.current) {
          const focusableElements = dialogRef.current.querySelectorAll(
            'button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])'
          );
          if (focusableElements.length > 0) focusableElements[0].focus();
        }
      }, 50);

      const handleKeyDown = (e) => {
        if (e.key === 'Escape') {
          e.preventDefault();
          onClose();
        } else if (e.key === 'Tab' && dialogRef.current) {
          const focusableElements = dialogRef.current.querySelectorAll(
            'button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])'
          );
          if (focusableElements.length === 0) return;
          const first = focusableElements[0];
          const last = focusableElements[focusableElements.length - 1];

          if (e.shiftKey && document.activeElement === first) {
            last.focus(); e.preventDefault();
          } else if (!e.shiftKey && document.activeElement === last) {
            first.focus(); e.preventDefault();
          }
        }
      };

      document.addEventListener('keydown', handleKeyDown);
      return () => {
        clearTimeout(focusTimer);
        document.removeEventListener('keydown', handleKeyDown);
        if (previousFocusRef.current && typeof previousFocusRef.current.focus === 'function') {
          previousFocusRef.current.focus();
        }
      };
    }
  }, [isOpen, onClose]);

  return dialogRef;
}

function useStoryMachine(nodes, reducedMotion) {
  const [machineState, setMachineState] = useState({ phase: 'idle', targetId: null, activeId: null });

  const navigateTo = useCallback((targetId) => {
    setMachineState(prev => {
      if (prev.activeId === targetId) return prev;
      if (prev.activeId) return { phase: 'pullback', activeId: null, targetId }; 
      return { phase: 'pushin', activeId: null, targetId }; 
    });
  }, []);

  const closeArticle = useCallback(() => {
    setMachineState(prev => {
      if (prev.phase === 'idle' && !prev.activeId) return prev;
      return { phase: 'pullback', activeId: null, targetId: null };
    });
  }, []);

  useEffect(() => {
    if (machineState.phase === 'idle') return;
    
    let duration = machineState.phase === 'pullback' ? CONFIG.TIMING.PULLBACK_DURATION : CONFIG.TIMING.PUSHIN_DURATION;
    if (reducedMotion) duration = 10; 

    const timerId = setTimeout(() => {
      setMachineState(prev => {
        if (prev.phase === 'pullback') return prev.targetId ? { ...prev, phase: 'pushin' } : { phase: 'idle', activeId: null, targetId: null };
        if (prev.phase === 'pushin') return { phase: 'idle', activeId: prev.targetId, targetId: null };
        return prev;
      });
    }, duration + 50);

    return () => clearTimeout(timerId);
  }, [machineState.phase, reducedMotion]);

  return { machineState, navigateTo, closeArticle };
}

                    <span className="text-[9px] md:text-[10px] text-neutral-500 font-light tracking-[0.3em] uppercase">下一站</span>
                    <span className="text-lg md:text-xl text-white/95 font-medium tracking-wide">{nextArticleNode.chapter}: {nextArticleNode.title}</span>
                  </div>
                  <div className="w-12 h-12 rounded-full border border-white/10 flex items-center justify-center group-hover:bg-white/10 transition-all">
                    <ArrowRight size={20} className="text-neutral-400 group-hover:text-white group-hover:translate-x-1.5 transition-all" strokeWidth={1} />
                  </div>
                </div>
              </button>
            ) : (
              <button 
                onClick={onClose} 
                className="flex items-center gap-4 mx-auto px-8 py-4 rounded-full border border-white/15 bg-black/40 text-[9px] tracking-[0.3em] text-white/70 hover:text-white transition-all focus-visible:ring-2 focus-visible:ring-white/50 focus:outline-none"
              >
                <RotateCcw size={14} strokeWidth={1.5} /> 返回軌道
              </button>
            )}
          </nav>
        </div>
      </article>
    </aside>
  );
};

const PlanetNode = ({ node, isHovered, isSelected, isTarget, phase, isReducedMotion, shouldBlur, onMouseEnter, onMouseLeave, onNavigate }) => {
  const rootRef = useRef(null);
  const bodyRef = useRef(null);
  const labelContainerRef = useRef(null);
  const labelTextRef = useRef(null);

  const engine = useContext(EngineContext);

  useEffect(() => {
    if (!engine) return;
    const unsubscribe = engine.subscribeNode(node.id, ({ x, y, z, scale, labelY, labelScale }) => {
      if (rootRef.current) rootRef.current.style.transform = `translate3d(${x}px, ${y}px, ${z}px) rotateX(-75deg)`;
      if (bodyRef.current) bodyRef.current.style.transform = `translate(-50%, -50%) scale(${scale})`;
      if (labelContainerRef.current) labelContainerRef.current.style.transform = `translate(-50%, ${labelY}px)`;
      if (labelTextRef.current) labelTextRef.current.style.transform = `scale(${labelScale})`;
    });
    return unsubscribe;
  }, [engine, node.id]);

  let transitionDuration;
  if (isReducedMotion) transitionDuration = '1ms';
  else if (phase === 'pullback') transitionDuration = `${CONFIG.TIMING.PULLBACK_DURATION}ms`;
  else if (phase === 'pushin') transitionDuration = `${CONFIG.TIMING.PUSHIN_DURATION}ms`;
  else transitionDuration = `${CONFIG.TIMING.DEFAULT_TRANSITION}ms`;

  const isZooming = phase !== 'idle';
  const isCenterStar = node.radius === 0;

  return (
    <div
      ref={rootRef}
      role="button"
      tabIndex={isZooming ? -1 : 0}
      aria-label={`檢視故事: ${node.chapter} ${node.title}`}
      className={`absolute top-0 left-0 preserve-3d cursor-pointer group focus:outline-none focus-visible:ring-2 focus-visible:ring-white/50 focus-visible:rounded-full
        ${!isCenterStar ? 'will-change-transform' : ''}
        ${shouldBlur ? 'opacity-10 blur-md pointer-events-none' : 'opacity-100 blur-0'}
      `}
      onMouseEnter={() => !isZooming && onMouseEnter(node.id)}
      onMouseLeave={() => !isZooming && onMouseLeave()}
      onClick={() => !isZooming && onNavigate(node.id)}
      onKeyDown={(e) => {
        if ((e.key === 'Enter' || e.key === ' ') && !isZooming) {
          e.preventDefault();
          onNavigate(node.id);
        }
      }}
      style={{ 
        transitionProperty: 'opacity, filter', 
        transitionDuration: transitionDuration,
        transitionTimingFunction: 'cubic-bezier(0.16, 1, 0.3, 1)'
      }}
    >
      {isCenterStar && !isReducedMotion && (
        <>
          <div className="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 w-[160%] h-[160%] rounded-full animate-pulse bg-orange-400/20 blur-[15px] pointer-events-none" />
          <div className="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 w-[250%] h-[250%] rounded-full bg-yellow-300/10 blur-[30px] pointer-events-none" />
        </>
      )}

      {node.hasRing && (
        <div 
          className="absolute top-1/2 left-1/2 rounded-full border-[1px] border-yellow-200/30"
          style={{ 
            width: node.size * 2.8, height: node.size * 2.8,
            transform: 'translate(-50%, -50%) rotateX(75deg)', 
            boxShadow: '0 0 15px rgba(234, 179, 8, 0.15) inset, 0 0 15px rgba(234, 179, 8, 0.15)'
          }} 
        />
      )}

      <div 
        ref={bodyRef}
        className={`absolute top-1/2 left-1/2 rounded-full bg-gradient-to-br ${node.color}`}
        style={{
          width: node.size, height: node.size,
          transitionProperty: 'box-shadow', 
          transitionDuration: transitionDuration,
          transitionTimingFunction: 'cubic-bezier(0.16, 1, 0.3, 1)',
          boxShadow: isCenterStar
            ? `0 0 40px 10px ${node.glow}, 0 0 80px 30px rgba(249, 115, 22, 0.2), inset 0 0 15px rgba(255,255,255,0.8)`
            : (isHovered || isSelected || (isTarget && isZooming))
              ? `0 0 45px 12px ${node.glow}, inset 0 0 15px rgba(255,255,255,0.9)`
              : `0 0 12px 1px ${node.glow}`
        }}
      />

      <div ref={labelContainerRef} className="absolute bottom-full left-1/2 mb-4 flex flex-col items-center pointer-events-none">
        <span 
          ref={labelTextRef} 
          className={`whitespace-nowrap text-[10px] sm:text-[11px] uppercase tracking-[0.25em] font-medium px-3 py-1.5 rounded-full border
          ${(isHovered || isSelected) && !isZooming
            ? 'text-white border-white/30 bg-white/10 backdrop-blur-md drop-shadow-[0_0_15px_rgba(255,255,255,0.8)]'
            : 'text-white/80 border-white/5 bg-[#020308]/60 backdrop-blur-md shadow-[0_4px_10px_rgba(0,0,0,0.5)]'}`}
            style={{
              transitionProperty: 'color, background-color, border-color, box-shadow',
              transitionDuration: transitionDuration,
              transitionTimingFunction: 'cubic-bezier(0.16, 1, 0.3, 1)'
            }}
        >
          {node.chapter}
        </span>
      </div>
    </div>
  );
};

export default function App() {
  const { windowWidth, scale } = useWindowContext();
  const isReducedMotion = useReducedMotion();
  const { machineState, navigateTo, closeArticle } = useStoryMachine(APP_DATA, isReducedMotion);
  
  const [hoveredPlanet, setHoveredPlanet] = useState(null);
  const [isUserPaused, setIsUserPaused] = useState(false);

  const engineRef = useRef(null);
  if (!engineRef.current) engineRef.current = new SolarPhysicsEngine(APP_DATA, CONFIG.TIMING, isReducedMotion);

  const sceneRef = useRef(null);

  useEffect(() => {
    engineRef.current.updateContext({
      isPaused: isUserPaused,
      hoveredId: hoveredPlanet,
      targetId: machineState.targetId,
      activeId: machineState.activeId,
      phase: machineState.phase
    }, isReducedMotion);
  }, [isUserPaused, hoveredPlanet, machineState, isReducedMotion]);

  useEffect(() => {
    engineRef.current.start();
    const unsubscribe = engineRef.current.subscribeCamera((camX, camY) => {
      if (sceneRef.current) sceneRef.current.style.transform = `translate3d(${camX}px, ${camY}px, 0)`;
    });
    return () => {
      unsubscribe();
      engineRef.current.stop();
    };
  }, []);

  const getCameraTransitionDuration = () => {
    if (isReducedMotion) return '1ms';
    if (machineState.phase === 'pullback') return `${CONFIG.TIMING.PULLBACK_DURATION}ms`;
    if (machineState.phase === 'pushin') return `${CONFIG.TIMING.PUSHIN_DURATION}ms`;
    return `${CONFIG.TIMING.DEFAULT_TRANSITION}ms`;
  };

  const getCameraTransform = () => {
    const { phase, activeId } = machineState;
    const isMobile = windowWidth < CONFIG.LAYOUT.MD_BREAKPOINT;
    
    if (phase === 'pullback') return `scale(${scale * CONFIG.CAMERA.PULLBACK.scale}) rotateX(${CONFIG.CAMERA.PULLBACK.rotateX}deg) translateY(${CONFIG.CAMERA.PULLBACK.y}px)`; 
    if (phase === 'pushin') return `scale(${scale * CONFIG.CAMERA.PUSHIN.scale}) rotateX(${CONFIG.CAMERA.PUSHIN.rotateX}deg) translateY(${CONFIG.CAMERA.PUSHIN.y}px)`; 
    
    if (activeId) {
      if (isMobile) return `scale(${scale * CONFIG.CAMERA.MOBILE_ACTIVE.scale}) rotateX(${CONFIG.CAMERA.MOBILE_ACTIVE.rotateX}deg) translateY(${CONFIG.CAMERA.MOBILE_ACTIVE.y}vh)`;
      const leftSpace = windowWidth - CONFIG.LAYOUT.PANEL_WIDTH;
      const safeScale = Math.max(0.35, Math.min(scale * 0.65, (leftSpace - 40) / CONFIG.LAYOUT.SYS_MAX_SIZE));
      return `scale(${safeScale}) rotateX(65deg) translateY(0px)`;
    }
    return `scale(${scale}) rotateX(${CONFIG.CAMERA.IDLE.rotateX}deg) translateY(0px)`;
  };

  return (
    <EngineContext.Provider value={engineRef.current}>
      <div className="w-full h-screen bg-[#020308] font-sans text-neutral-200 overflow-hidden relative selection:bg-neutral-800">
        
        <Background />
        <Header />

        <main 
          className="absolute inset-0 flex items-center justify-center perspective-[1200px] ease-[cubic-bezier(0.16,1,0.3,1)]" 
          style={{ 
            transform: machineState.activeId && machineState.phase === 'idle' && windowWidth >= CONFIG.LAYOUT.MD_BREAKPOINT 
              ? `translateX(-${CONFIG.LAYOUT.PANEL_WIDTH / 2}px)` 
              : `translateX(0px)`,
            transitionDuration: isReducedMotion ? '1ms' : `${CONFIG.TIMING.DEFAULT_TRANSITION}ms`,
            transitionProperty: 'transform'
          }}
        >
          <div 
            className="relative preserve-3d will-change-transform"
            style={{ 
              width: 0, height: 0, 
              transform: getCameraTransform(),
              transition: `transform ${getCameraTransitionDuration()} cubic-bezier(0.65, 0, 0.35, 1)` 
            }}
          >
            <div ref={sceneRef} className="absolute inset-0 preserve-3d will-change-transform">
              {APP_DATA.filter(n => n.radius > 0).map((node) => (
                <div
                  key={`orbit-${node.id}`}
                  className={`absolute top-0 left-0 rounded-full border border-white/[0.05] pointer-events-none transition-opacity ${
                    machineState.targetId && machineState.phase === 'pushin' ? 'opacity-0' : 'opacity-100'
                  }`}
                  style={{ 
                    width: node.radius * 2, height: node.radius * 2, transform: `translate(-50%, -50%)`,
                    transitionDuration: isReducedMotion ? '1ms' : '1000ms'
                  }}
                />
              ))}

              {APP_DATA.map((node) => (
                <PlanetNode
                  key={node.id}
                  node={node}
                  isHovered={hoveredPlanet === node.id}
                  isSelected={machineState.activeId === node.id}
                  isTarget={machineState.targetId === node.id}
                  phase={machineState.phase}
                  isReducedMotion={isReducedMotion}
                  shouldBlur={machineState.targetId !== null && machineState.targetId !== node.id && machineState.phase === 'pushin'}
                  onMouseEnter={setHoveredPlanet}
                  onMouseLeave={() => setHoveredPlanet(null)}
                  onNavigate={navigateTo}
                />
              ))}
            </div>
          </div>
        </main>

        <ArticlePanel 
          activeId={machineState.activeId} 
          isIdle={machineState.phase === 'idle'}
          onNavigate={navigateTo}
          onClose={closeArticle}
          nodes={APP_DATA}
          isReducedMotion={isReducedMotion}
        />
        
        <PlayControls 
          isUserPaused={isUserPaused} 
          togglePause={() => setIsUserPaused(!isUserPaused)} 
          isReducedMotion={isReducedMotion}
        />

      </div>
    </EngineContext.Provider>
  );
}
