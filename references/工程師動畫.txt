    <script>
        // 核心配置與環境偵測
        const AppConfig = {
            hasReducedMotion: window.matchMedia("(prefers-reduced-motion: reduce)").matches,
            checkDependencies: () => typeof window.gsap !== 'undefined' 
                                  && typeof window.ScrollTrigger !== 'undefined'
                                  && typeof window.Lenis !== 'undefined'
        };

        class SmoothScroll {
            constructor() {
                this.lenis = new Lenis({
                    duration: 1.2,
                    easing: (t) => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
                    direction: 'vertical',
                    smooth: true,
                    smoothTouch: false, // 行動裝置採用原生滾動，避免與 Touch Event 衝突造成抖動
                });

                this.init();
            }

            init() {
                this.lenis.on('scroll', ScrollTrigger.update);
                gsap.ticker.add((time) => this.lenis.raf(time * 1000));
                gsap.ticker.lagSmoothing(0);
            }
        }

        class ElectricTextAnimation {
            constructor(options) {
                this.triggerId = options.trigger;
                this.init();
            }

            init() {
                const tl = gsap.timeline({
                    scrollTrigger: {
                        trigger: this.triggerId,
                        start: "top top", 
                        end: "+=50%",          
                        pin: true,              
                        scrub: 1,             
                        anticipatePin: 1        
                    }
                });

                tl.fromTo(`${this.triggerId} .svg-wrapper`, 
                    { scale: 0.95 }, { scale: 1.05, duration: 5, ease: "none" }, 0
                );
                tl.to(`${this.triggerId} .svg-text-zap`, { opacity: 1, duration: 0.25 }, 0.25);
                tl.to(`${this.triggerId} .svg-text-stroke`, { strokeDashoffset: 0, duration: 3, ease: "power3.inOut" }, 0.5);
                tl.to(`${this.triggerId} .svg-text-zap`, { strokeDashoffset: -1500, duration: 3, ease: "power3.inOut" }, 0.5);
                
                tl.to(`${this.triggerId} .svg-text-zap`, { opacity: 0, duration: 0.4 }, 3.1);
                tl.to(`${this.triggerId} .svg-text-fill`, { opacity: 1, duration: 0.8, ease: "power2.inOut" }, 3.3);
                tl.to(`${this.triggerId} .svg-text-stroke`, { opacity: 0, duration: 0.8, ease: "power2.inOut" }, 3.3);
                tl.to(`${this.triggerId} .ambient-glow`, { opacity: 1, duration: 0.8, ease: "power2.inOut" }, 3.3); 
            }
        }

        /**
         * 系統啟動器 (Bootstrapper)
         * 負責狀態管理、防呆機制與依賴驗證
         */
        function bootstrapApp() {
            // 防禦 1: 處理瀏覽器重整時的 Scroll 錯位問題
            if ('scrollRestoration' in history) {
                history.scrollRestoration = 'manual';
            }

            // 防禦 2: 使用者偏好檢測
            if (AppConfig.hasReducedMotion) {
                console.info("[App] Reduced motion preferred. Bypassing animation enhancements.");
                return; 
            }

            // 防禦 3: 外部腳本載入驗證
            if (!AppConfig.checkDependencies()) {
                console.warn("[App] Critical dependencies failed to load. Graceful degradation applied.");
                return;
            }

            try {
                // 註冊核心套件
                gsap.registerPlugin(ScrollTrigger);

                // 防禦 4: 防止行動裝置網址列收合觸發 resize 導致 Pin 跳動
                ScrollTrigger.config({ ignoreMobileResize: true });

                // 標記應用程式就緒，讓 CSS 隱藏預設靜態文字並啟用特效準備
                document.documentElement.setAttribute('data-gsap-ready', 'true');

                // 實例化模組
                new SmoothScroll();
                new ElectricTextAnimation({ trigger: "#hero-section" });

                // 防禦 5: 確保所有字體與佈局載入完畢後，強制刷新 ScrollTrigger 計算
                window.addEventListener('load', () => ScrollTrigger.refresh());

            } catch (error) {
                console.error("[App] Initialization crashed. Reverting to fallback UI.", error);
                // 發生非預期錯誤，立刻拔除狀態，無縫退回安全降級模式
                document.documentElement.removeAttribute('data-gsap-ready');
            }
        }

        // 使用 DOMContentLoaded 確保 DOM 樹已建立，且 defer 腳本皆已執行完畢
        document.addEventListener("DOMContentLoaded", bootstrapApp);
    </script>
